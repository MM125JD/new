def get_google_services():
    creds = None
    if os.path.exists('token.json'):
        creds = Credentials.from_authorized_user_file('token.json', SCOPES)
    if not creds or not creds.valid:
        if creds and creds.expired and creds.refresh_token:
            creds.refresh(Request())
        else:
            flow = InstalledAppFlow.from_client_secrets_file('credentials.json', SCOPES)
            creds = flow.run_local_server(port=0)
        with open('token.json', 'w') as token:
            token.write(creds.to_json())
    return build('docs', 'v1', credentials=creds), build('drive', 'v3', credentials=creds)

def get_high_clap_articles():
    target_urls = ["https://medium.com/feed/tag/personal-growth", "https://medium.com/feed/tag/psychology"]
    valid_articles = []
    for url in target_urls:
        print(f"正在扫描 Medium 话题源: {url}...")
        try:
            response = requests.get(url, headers=headers, timeout=10)
            soup = BeautifulSoup(response.text, 'xml')
            for item in soup.find_all('item'):
                title = item.title.text
                link = item.link.text
                article_detail = requests.get(link, headers=headers, timeout=10)
                detail_soup = BeautifulSoup(article_detail.text, 'html.parser')
                paragraphs = detail_soup.find_all('p')
                if len(paragraphs) > 5:
                    valid_articles.append({"title": title, "content": "\n\n".join([p.text for p in paragraphs])})
                if len(valid_articles) >= 6:  # 顺便帮你多拿几篇
                    break
        except Exception as e:
            print(f"扫描源 {url} 失败: {e}")
    return valid_articles

def translate_all_at_once(articles):
    """将所有文章打包，一次性发送给 Gemini 翻译"""
    client = genai.Client(api_key=GEMINI_API_KEY)
    
    # 动态组装超级长文本
    combined_body = ""
    for idx, art in enumerate(articles, 1):
        combined_body += f"\n\n=========================================\n"
        combined_body += f"【第 {idx} 篇】文章标题: {art['title']}\n"
        combined_body += f"=========================================\n\n"
        combined_body += art['content']
        
    prompt = f"""你是一个精通心理学和个人成长领域的资深翻译家。
请将以下几篇英文文章翻译成高质量的中文。请务必完整保留我给出的原文章序号、分隔线和标题结构，行文要流畅，优雅且富有启发性：

{combined_content if 'combined_content' in locals() else combined_body}"""

    response = client.models.generate_content(
        model="gemini-2.0-flash",
        contents=prompt
    )
    return response.text

def create_combined_google_doc(docs_service, content):
    """在 Google 云端硬盘创建当天的合集文档"""
    import datetime
    today_str = datetime.date.today().strftime("%Y-%m-%d")
    body = {'title': f"[Medium优质文章翻译精选合集] - {today_str}"}
    doc = docs_service.documents().create(body=body).execute()
    doc_id = doc.get('documentId')

    requests_body = [{'insertText': {'location': {'index': 1}, 'text': content}}]
    docs_service.documents().batchUpdate(documentId=doc_id, body={'requests': requests_body}).execute()
    print(f"✨ 奇迹时刻！全量文章已成功一次性同步至谷歌文档: {body['title']}")
