# 🎨 极客色彩调色盘

拉动滑块调配你的专属主题色，双击 Hex 代码可以直接复制：

<script setup>
import { ref, computed } from 'vue'

const r = ref(180)
const g = ref(190)
const b = ref(254)

// 动态计算出 RGB 格式
const rgbString = computed(() => `rgb(${r.value}, ${g.value}, ${b.value})`)

// 动态计算出 16 进制 Hex 格式
const hexString = computed(() => {
  const toHex = (c) => c.toString(16).padStart(2, '0')
  return `#${toHex(r.value)}${toHex(g.value)}${toHex(b.value)}`.toUpperCase()
})

const copyToClipboard = () => {
  navigator.clipboard.writeText(hexString.value)
  alert(`复制成功: ${hexString.value} 🚀`)
}
</script>

<div :style="{ backgroundColor: rgbString }" style="padding: 30px; border-radius: 12px; transition: background-color 0.2s ease; margin: 20px 0; text-align: center; border: 1px solid var(--vp-c-border);">
  
  <!-- 颜色指标卡片 -->
  <div style="background: rgba(0, 0, 0, 0.6); color: #fff; padding: 15px; display: inline-block; border-radius: 8px; backdrop-filter: blur(4px);">
    <p style="margin: 0; font-family: monospace; font-size: 1.5rem; cursor: pointer;" @dblclick="copyToClipboard" title="双击复制">
      Hex: {{ hexString }} (双击复制)
    </p>
    <p style="margin: 5px 0 0 0; font-size: 0.9rem; opacity: 0.8;">{{ rgbString }}</p>
  </div>

  <!-- 滑块控制区 -->
  <div style="margin-top: 25px; background: rgba(255,255,255,0.9); padding: 15px; border-radius: 8px; color: #111; max-width: 350px; margin-left: auto; margin-right: auto; text-align: left;">
    <div style="margin-bottom: 10px;">
      <span style="display:inline-block; width: 30px; font-weight: bold; color: #ff757f;">R:</span>
      <input type="range" min="0" max="255" v-model.number="r" style="width: 80%;" />
      <span style="font-family: monospace; margin-left: 10px;">{{ r }}</span>
    </div>
    <div style="margin-bottom: 10px;">
      <span style="display:inline-block; width: 30px; font-weight: bold; color: #c3e88d;">G:</span>
      <input type="range" min="0" max="255" v-model.number="g" style="width: 80%;" />
      <span style="font-family: monospace; margin-left: 10px;">{{ g }}</span>
    </div>
    <div>
      <span style="display:inline-block; width: 30px; font-weight: bold; color: #82aaff;">B:</span>
      <input type="range" min="0" max="255" v-model.number="b" style="width: 80%;" />
      <span style="font-family: monospace; margin-left: 10px;">{{ b }}</span>
    </div>
  </div>
</div>
