# ⏳ 极客专注时钟

在阅读我的技术笔记时，不妨开启一个 25 分钟的番茄专注：

<script setup>
import { ref, onUnmounted } from 'vue'

const timeLeft = ref(25 * 60) // 25分钟
const isRunning = ref(false)
let timerId = null

const formatTime = (seconds) => {
  const m = Math.floor(seconds / 60).toString().padStart(2, '0')
  const s = (seconds % 60).toString().padStart(2, '0')
  return `${m}:${s}`
}

const toggleTimer = () => {
  if (isRunning.value) {
    clearInterval(timerId)
    isRunning.value = false
  } else {
    isRunning.value = true
    timerId = setInterval(() => {
      if (timeLeft.value > 0) {
        timeLeft.value--
      } else {
        clearInterval(timerId)
        isRunning.value = false
        alert('🎉 恭喜你完成了一次专注！')
      }
    }, 1000)
  }
}

const resetTimer = () => {
  clearInterval(timerId)
  isRunning.value = false
  timeLeft.value = 25 * 60
}

onUnmounted(() => clearInterval(timerId))
</script>

<div style="text-align: center; padding: 20px; border: 2px dashed #b4befe; border-radius: 12px; margin: 20px 0;">
  <div style="font-size: 3rem; font-family: monospace; font-weight: bold; color: #b4befe; margin-bottom: 10px;">
    {{ formatTime(timeLeft) }}
  </div>
  <button @click="toggleTimer" style="background: #a6e3a1; color: #11111b; padding: 6px 16px; border-radius: 6px; font-weight: bold; margin-right: 10px;">
    {{ isRunning ? '暂停' : '开始专注' }}
  </button>
  <button @click="resetTimer" style="background: #f38ba8; color: #11111b; padding: 6px 16px; border-radius: 6px; font-weight: bold;">
    重置
  </button>
</div>
