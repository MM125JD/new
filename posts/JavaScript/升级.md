# 🗣️ 极客多国语言朗读器 2.0

选择你喜欢的音色，输入你想听的语言（支持中文、英文、韩语等），点击播放：

<script setup>
import { ref, onMounted, onUnmounted } from 'vue';

const textToSpeak = ref('안녕하세요! 한국어 공부를 응원합니다. 화이팅!') // 默认放一句暖心的韩语
const isSpeaking = ref(false)
const voices = ref([])
const selectedVoiceName = ref('')

// 1. 动态加载系统和浏览器支持的全部声音列表
const loadVoices = () => {
  if (typeof window !== 'undefined' && 'speechSynthesis' in window) {
    const allVoices = window.speechSynthesis.getVoices()
    voices.value = allVoices
    
    // 如果还没选择声音，默认选一个韩语(ko-KR)或中文(zh-CN)的声音
    if (!selectedVoiceName.value && allVoices.length > 0) {
      const defaultVoice = allVoices.find(v => v.lang.includes('ko') || v.lang.includes('zh') || v.lang.includes('en')) || allVoices[0]
      selectedVoiceName.value = defaultVoice.name
    }
  }
}

onMounted(() => {
  loadVoices()
  if (typeof window !== 'undefined' && 'speechSynthesis' in window) {
    // 谷歌浏览器（Chrome）等浏览器是异步加载声音的，必须监听这个事件
    window.speechSynthesis.onvoiceschanged = loadVoices
  }
})

// 2. 快捷输入预设文本
const setPreset = (lang) => {
  if (lang === 'ko') {
    textToSpeak.value = '안녕하세요! 오늘 하루도 좋은 하루 보내세요.' // 韩语：你好！今天也度过美好的一天吧。
  } else if (lang === 'en') {
    textToSpeak.value = 'Keep trying, no matter how hard it seems. You will get there.' // 英语励志
  } else if (lang === 'zh') {
    textToSpeak.value = '精益求精，每天进步一点点，这就是极客精神！' // 中文
  }
}

// 3. 播放控制
const speak = () => {
  if ('speechSynthesis' in window) {
    window.speechSynthesis.cancel() // 播放前先停止之前的声音
    
    const utterance = new SpeechSynthesisUtterance(textToSpeak.value)
    
    // 绑定选中的声音
    const activeVoice = voices.value.find(v => v.name === selectedVoiceName.value)
    if (activeVoice) {
      utterance.voice = activeVoice
      utterance.lang = activeVoice.lang // 自动同步语言设定
    }
    
    utterance.onstart = () => isSpeaking.value = true
    utterance.onend = () => isSpeaking.value = false
    utterance.onerror = () => isSpeaking.value = false
    
    window.speechSynthesis.speak(utterance)
  } else {
    alert('抱歉，您的浏览器不支持语音合成功能。')
  }
}

const stop = () => {
  window.speechSynthesis.cancel()
  isSpeaking.value = false
}

onUnmounted(() => {
  if (typeof window !== 'undefined' && 'speechSynthesis' in window) {
    window.speechSynthesis.cancel()
  }
})
</script>

<div style="padding: 20px; border: 1px solid var(--vp-c-border); border-radius: 12px; margin: 20px 0; background: var(--vp-c-bg-alt);">

  <!-- 快捷语种预设 -->
  <div style="margin-bottom: 12px; display: flex; gap: 8px; align-items: center;">
    <span style="font-size: 0.9rem; color: var(--vp-c-text-2);">快捷输入：</span>
    <button @click="setPreset('ko')" style="background: #f9e2af; color: #11111b; padding: 2px 10px; border-radius: 4px; font-size: 0.8rem; font-weight: bold; border: none; cursor: pointer;">🇰🇷 韩语预设</button>
    <button @click="setPreset('en')" style="background: #a6e3a1; color: #11111b; padding: 2px 10px; border-radius: 4px; font-size: 0.8rem; font-weight: bold; border: none; cursor: pointer;">🇺🇸 英语预设</button>
    <button @click="setPreset('zh')" style="background: #89b4fa; color: #11111b; padding: 2px 10px; border-radius: 4px; font-size: 0.8rem; font-weight: bold; border: none; cursor: pointer;">🇨🇳 中文预设</button>
  </div>

  <!-- 文本输入区 -->
<textarea
v-model="textToSpeak"
rows="3"
style="width: 100%; padding: 12px; border-radius: 8px; border: 1px solid #ccc; background: var(--vp-c-bg); color: var(--vp-c-text-1); font-size: 1rem; resize: none; line-height: 1.5;"
placeholder="在此输入想要朗读的内容...">
</textarea>

  <!-- 声音/语言选择器 -->
  <div style="margin-top: 15px; display: flex; flex-direction: column; gap: 8px;">
    <label style="font-size: 0.9rem; font-weight: bold; color: var(--vp-c-text-1);">选择音色（包含系统及浏览器内所有声音）：</label>
    <select 
      v-model="selectedVoiceName" 
      style="width: 100%; padding: 8px; border-radius: 6px; border: 1px solid #ccc; background: var(--vp-c-bg); color: var(--vp-c-text-1); font-size: 0.9rem;"
    >
      <option v-for="voice in voices" :key="voice.name" :value="voice.name">
        {{ voice.name }} ({{ voice.lang }}) {{ voice.localService ? '[本地]' : '[云端]' }}
      </option>
    </select>
  </div>

  <!-- 控制按钮 -->
  <div style="margin-top: 20px; display: flex; gap: 10px;">
    <button @click="speak" :style="{ background: isSpeaking ? '#a6e3a1' : '#b4befe' }" style="color: #11111b; padding: 10px 24px; border-radius: 8px; font-weight: bold; border: none; cursor: pointer; display: flex; align-items: center; gap: 6px;">
      {{ isSpeaking ? '🔊 正在朗读...' : '▶️ 开始朗读' }}
    </button>
    <button v-if="isSpeaking" @click="stop" style="background: #f38ba8; color: #11111b; padding: 10px 24px; border-radius: 8px; font-weight: bold; border: none; cursor: pointer;">
      🛑 停止
    </button>
  </div>
</div>
