# 📌 浏览器 local 贴纸

写下你当下瞬时的灵感或待办事项，刷新网页也不会丢失：

<script setup>
import { ref, watch, onMounted } from 'vue';

const memoText = ref('')

// 1. 页面加载时，从浏览器的 LocalStorage 读取之前保存的值
onMounted(() => {
  const savedMemo = localStorage.getItem('local_memo_data')
  if (savedMemo) {
    memoText.value = savedMemo
  }
})

// 2. 监听 memoText 的变化，只要输入框变了，自动保存到本地，无需手动点保存按钮
watch(memoText, (newVal) => {
  localStorage.setItem('local_memo_data', newVal)
})

const clearMemo = () => {
  if (confirm('确定要清空当前的贴纸内容吗？')) {
    memoText.value = ''
  }
}
</script>

<div style="padding: 20px; border: 1px solid #f9e2af; background: #f9e2af22; border-radius: 12px; margin: 20px 0; position: relative;">
  <div style="position: absolute; top: -10px; left: 15px; background: #f9e2af; color: #11111b; padding: 2px 8px; border-radius: 4px; font-size: 0.8rem; font-weight: bold;">
    LOCAL MEMO (本地数据)
  </div>
  
  <textarea 
    v-model="memoText"
    style="width: 100%; min-height: 120px; background: transparent; border: none; outline: none; color: var(--vp-c-text-1); font-family: 'Courier New', Courier, monospace; font-size: 1.05rem; line-height: 1.6; resize: vertical; margin-top: 10px;"
    placeholder="在此输入一些随手记的内容，我会静静地帮保存在本地...">
  </textarea>
  
  <div style="text-align: right; margin-top: 10px;">
    <button @click="clearMemo" style="background: transparent; color: #f38ba8; border: 1px solid #f38ba8; padding: 3px 10px; border-radius: 4px; font-size: 0.8rem; cursor: pointer;">
      🗑️ 清空
    </button>
  </div>
</div>
