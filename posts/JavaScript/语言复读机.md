# 🗣️ 语音复读机 (Web Speech API)

输入一段文字（比如你正在学的英语句子），让浏览器朗读出来：

<script setup>
import { ref } from 'vue'

const textToSpeak = ref('Hello! Keep studying and happy coding every day.')
const isSpeaking = ref(false)

const speak = () => {
  if ('speechSynthesis' in window) {
    // 如果正在朗读，先停止
    window.speechSynthesis.cancel()
    
    const utterance = new SpeechSynthesisUtterance(textToSpeak.value)
    // 自动识别语言，也可以手动设定如 utterance.lang = 'en-US'
    
    utterance.onstart = () => isSpeaking.value = true
    utterance.onend = () => isSpeaking.value = false
    
    window.speechSynthesis.speak(utterance)
  } else {
    alert('抱歉，您的浏览器不支持语音合成功能功能。')
  }
}

const stop = () => {
  window.speechSynthesis.cancel()
  isSpeaking.value = false
}
</script>

<div style="padding: 20px; border: 1px solid var(--vp-c-border); border-radius: 8px; margin: 20px 0; background: var(--vp-c-bg-alt);">
  <textarea 
    v-model="textToSpeak" 
    rows="3" 
    style="width: 100%; padding: 10px; border-radius: 6px; border: 1px solid #ccc; background: var(--vp-c-bg); color: var(--vp-c-text-1); font-size: 1rem; resize: none;"
    placeholder="在此输入想要朗读的内容..."
  ></textarea>
  
  <div style="margin-top: 15px; display: flex; gap: 10px;">
    <button @click="speak" :style="{ background: isSpeaking ? '#a6e3a1' : '#b4befe' }" style="color: #11111b; padding: 8px 20px; border-radius: 6px; font-weight: bold; border: none; cursor: pointer;">
      {{ isSpeaking ? '🔊 正在朗读...' : '▶️ 播放声音' }}
    </button>
    <button v-if="isSpeaking" @click="stop" style="background: #f38ba8; color: #11111b; padding: 8px 20px; border-radius: 6px; font-weight: bold; border: none; cursor: pointer;">
      🛑 停止
    </button>
  </div>
</div>
