<script setup>
import { withBase } from 'vitepress'
import { computed, ref, onMounted } from 'vue'

// 💡 1. 导入你的数据（请根据实际位置调整路径）
import { data as allData } from './.vitepress/theme/posts.data.js' 

// 💡 2. 过滤并分组
const ssPosts = computed(() => {
  return allData.filter(post => post.url.includes('/ss/'))
})

const groupedPosts = computed(() => {
  const groups = {}
  ssPosts.value.forEach(post => {
    const cat = post.category || '未分类'
    if (!groups[cat]) {
      groups[cat] = []
    }
    groups[cat].push(post)
  })
  return groups
})

// 💡 3. ================= 文件夹大厅分页逻辑 =================
const currentFolderPage = ref(1) // 文件夹当前页码
const folderPageSize = 9         // 电脑端一页显示 9 个

// 将归类好的对象转为数组方便切片: [{ name: '上的', posts: [...] }, ...]
const folderList = computed(() => {
  return Object.entries(groupedPosts.value).map(([name, posts]) => ({
    name,
    posts
  }))
})

// 计算文件夹总页数
const totalFolderPages = computed(() => {
  return Math.ceil(folderList.value.length / folderPageSize) || 1
})

// 过滤出当前页应该显示的 9 个文件夹
const paginatedFolders = computed(() => {
  const startIndex = (currentFolderPage.value - 1) * folderPageSize
  const endIndex = startIndex + folderPageSize
  return folderList.value.slice(startIndex, endIndex)
})

// 文件夹翻页方法
const prevFolderPage = () => {
  if (currentFolderPage.value > 1) {
    currentFolderPage.value--
  }
}

const nextFolderPage = () => {
  if (currentFolderPage.value < totalFolderPages.value) {
    currentFolderPage.value++
  }
}


// 💡 4. ================= 文章列表分页逻辑 =================
const currentCategory = ref(null)
const currentPage = ref(1) // 文章列表当前页码
const pageSize = 10       // 页面二一页限制显示 10 篇

// 从 URL 解析参数
const parseCategoryFromUrl = () => {
  if (typeof window !== 'undefined') {
    const params = new URLSearchParams(window.location.search)
    currentCategory.value = params.get('cat')
    currentPage.value = 1 // 切换分类时重置文章页码
  }
}

onMounted(() => {
  parseCategoryFromUrl()
  window.addEventListener('popstate', parseCategoryFromUrl)
})

// 💡 修改后的 goCategory 函数
const goCategory = (catName) => {
  // 💡 设置你自己的密码，这里以 "123456" 为例
  const password = "123456"; 
  
  // 弹出输入框
  const input = prompt("请输入访问密码：");

  if (input === password) {
    // 密码正确，执行原有的逻辑
    currentCategory.value = catName;
    if (typeof window !== 'undefined') {
      const newUrl = `${window.location.pathname}?cat=${encodeURIComponent(catName)}`;
      window.history.pushState({ path: newUrl }, '', newUrl);
    }
  } else if (input !== null) {
    // 用户点了确定但密码错误
    alert("密码错误，无法访问！");
  }
  // 如果用户点了取消，直接忽略不做处理
};

const goBack = () => {
  currentCategory.value = null
  if (typeof window !== 'undefined') {
    window.history.pushState({ path: window.location.pathname }, '', window.location.pathname)
  }
}

// 获取当前选中分类下的文章总数
const totalPosts = computed(() => {
  if (!currentCategory.value || !groupedPosts.value[currentCategory.value]) return 0
  return groupedPosts.value[currentCategory.value].length
})

// 计算文章总页数
const totalPages = computed(() => {
  return Math.ceil(totalPosts.value / pageSize) || 1
})

// 截取当前页要显示的文章
const paginatedPosts = computed(() => {
  if (!currentCategory.value || !groupedPosts.value[currentCategory.value]) return []
  const startIndex = (currentPage.value - 1) * pageSize
  const endIndex = startIndex + pageSize
  return groupedPosts.value[currentCategory.value].slice(startIndex, endIndex)
})

const prevPage = () => {
  if (currentPage.value > 1) {
    currentPage.value--
    scrollToTop()
  }
}

const nextPage = () => {
  if (currentPage.value < totalPages.value) {
    currentPage.value++
    scrollToTop()
  }
}

const scrollToTop = () => {
  window.scrollTo({ top: 0, behavior: 'smooth' })
}
</script>
